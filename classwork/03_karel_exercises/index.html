      • findNextFence() рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃА рЃфрЃўрЃЎрЃџрЃў рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃгрЃДрЃЊрЃћрЃЉрЃљ, рЃЎрЃљрЃарЃћрЃџрЃў рЃЏрЃљрЃўрЃюрЃф рЃљрЃњрЃарЃФрЃћрЃџрЃћрЃЉрЃА рЃЏрЃЌрЃљрЃЋрЃљрЃарЃў рЃфрЃўрЃЎрЃџрЃўрЃА рЃерЃћрЃАрЃарЃБрЃџрЃћрЃЉрЃљрЃА, рЃљрЃюрЃБ рЃљрЃЏ рЃцрЃБрЃюрЃЦрЃфрЃўрЃљрЃерЃў рЃљрЃарЃљрЃљ рЃърЃарЃЮрЃЉрЃџрЃћрЃЏрЃљ
    • рЃЌрЃБ jump()-рЃўрЃА рЃерЃћрЃЏрЃЊрЃћрЃњ рЃћрЃњрЃарЃћрЃЋрЃћ рЃдрЃЮрЃЉрЃћрЃљ, рЃЏрЃЌрЃљрЃЋрЃљрЃарЃў while (frontIsClear()) рЃърЃўрЃарЃЮрЃЉрЃљ рЃљрЃдрЃљрЃа рЃерЃћрЃАрЃарЃБрЃџрЃЊрЃћрЃЉрЃљ рЃЊрЃљ рЃЎрЃљрЃарЃћрЃџрЃў рЃЊрЃљрЃљрЃАрЃарЃБрЃџрЃћрЃЉрЃА рЃЏрЃБрЃерЃљрЃЮрЃЉрЃљрЃА.

рЃњрЃљрЃЏрЃЮрЃАрЃљрЃЋрЃљрЃџрЃў рЃљрЃарЃўрЃА рЃАрЃ«рЃЋрЃљрЃюрЃљрЃўрЃарЃў рЃърЃўрЃарЃЮрЃЉрЃўрЃА рЃерЃћрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃљ. рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃљрЃЊ, рЃЌрЃБ рЃњрЃљрЃЊрЃљрЃ«рЃбрЃЮрЃЏрЃўрЃАрЃљрЃА рЃЏрЃћрЃЮрЃарЃћ рЃБрЃ»рЃарЃљрЃќрЃћ рЃЎрЃћрЃЊрЃћрЃџрЃў рЃЊрЃљрЃњрЃ«рЃЋрЃЊрЃљ, рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃЊрЃљрЃбрЃарЃўрЃљрЃџрЃћрЃЉрЃљ рЃЊрЃљ рЃБрЃЎрЃљрЃю рЃгрЃљрЃАрЃЋрЃџрЃљ рЃАрЃљрЃћрЃарЃЌрЃЮрЃЊ рЃљрЃа рЃњрЃљрЃЋрЃљрЃЎрЃћрЃЌрЃЮрЃЌ, рЃарЃљ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљрЃерЃўрЃф рЃерЃћрЃЏрЃЊрЃћрЃњ рЃфрЃўрЃЎрЃџрЃќрЃћ рЃЎрЃљрЃарЃћрЃџрЃўрЃА рЃЏрЃўрЃЏрЃљрЃарЃЌрЃБрЃџрЃћрЃЉрЃљ рЃЕрЃарЃЊрЃўрЃџрЃЮрЃћрЃЌрЃў рЃўрЃЦрЃюрЃћрЃЉрЃљ.

function start() {
    while(facingEast()) {
        findNextHurdle();
        jump();
    }
    // рЃљрЃЏ рЃфрЃўрЃЎрЃџрЃўрЃА рЃЊрЃљрЃАрЃарЃБрЃџрЃћрЃЉрЃўрЃАрЃљрЃА рЃЎрЃљрЃарЃћрЃџрЃў 
    // рЃљрЃарЃўрЃА рЃЉрЃЮрЃџрЃЮ рЃБрЃ»рЃарЃўрЃА рЃЏрЃћрЃЮрЃарЃћ рЃарЃўрЃњрЃерЃў,
    // рЃўрЃДрЃБрЃарЃћрЃЉрЃљ рЃЕрЃарЃЊрЃўрЃџрЃЮрЃћрЃЌрЃўрЃЌ
    turnAround();
    move();
    turnLeft();
}

function jump() {
    turnLeft();
    move();
    if (rightIsClear()) {
        turnRight();
        move();
        turnRight();
        move();
        turnLeft();
    }
}

5. рЃЏрЃљрЃарЃбрЃўрЃЋрЃў рЃљрЃЏрЃЮрЃ«рЃАрЃюрЃљ

function start() {
    for (var i = 0; i < 12; i++) {
        if (frontIsClear()) {
            move();
        } else {
            jump();
        }
    }
}
