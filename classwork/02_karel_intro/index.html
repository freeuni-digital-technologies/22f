рЃЎрЃЮрЃЊрЃў, рЃарЃЮрЃЏрЃћрЃџрЃАрЃљрЃф рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ рЃЊрЃљрЃљрЃўрЃњрЃюрЃЮрЃарЃћрЃЉрЃА. рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃЮрЃЌ рЃљрЃ«рЃАрЃюрЃћрЃЉрЃўрЃА рЃЊрЃљрЃАрЃљрЃЏрЃљрЃбрЃћрЃЉрЃџрЃљрЃЊ. 

takeBall(); // рЃћрЃарЃЌрЃ«рЃљрЃќрЃўрЃљрЃюрЃў рЃЎрЃЮрЃЏрЃћрЃюрЃбрЃљрЃарЃў. 

/*
рЃарЃљрЃЏрЃЊрЃћрЃюрЃўрЃЏрЃћ рЃ«рЃљрЃќрЃўрЃљрЃюрЃў рЃЎрЃЮрЃЏрЃћрЃюрЃбрЃљрЃарЃў
рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃљрЃЊ рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃА рЃгрЃўрЃю рЃљрЃЏ рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃА рЃљрЃ«рЃАрЃюрЃўрЃА рЃЊрЃљрЃАрЃљрЃгрЃћрЃарЃљрЃЊ
*/

рЃўрЃюрЃЊрЃћрЃюрЃбрЃљрЃфрЃўрЃљ

рЃЎрЃЮрЃЊрЃўрЃА рЃЉрЃџрЃЮрЃЎрЃћрЃЉрЃА рЃцрЃБрЃюрЃЦрЃфрЃўрЃћрЃЉрЃўрЃА рЃњрЃљрЃарЃЊрЃљ рЃАрЃ«рЃЋрЃљ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃўрЃф рЃўрЃДрЃћрЃюрЃћрЃЉрЃћрЃю рЃЊрЃљ рЃЕрЃЋрЃћрЃю рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃљрЃЏ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃўрЃА рЃЎрЃЮрЃЏрЃЉрЃўрЃюрЃљрЃфрЃўрЃљ (рЃћрЃарЃЌ рЃЉрЃџрЃЮрЃЎрЃерЃў рЃЏрЃћрЃЮрЃарЃўрЃА рЃњрЃљрЃ«рЃАрЃюрЃљ рЃЊрЃљ рЃљ.рЃе). рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃюрЃўрЃљ, рЃарЃЮрЃЏ рЃљрЃЏ рЃЊрЃарЃЮрЃА рЃЊрЃљрЃўрЃфрЃЋрЃљрЃЌ рЃўрЃюрЃЊрЃћрЃюрЃбрЃљрЃфрЃўрЃљ рЃЊрЃљ рЃДрЃЮрЃЋрЃћрЃџрЃў рЃљрЃ«рЃљрЃџрЃў рЃЉрЃџрЃЮрЃЎрЃўрЃА рЃ«рЃљрЃќрЃћрЃЉрЃў рЃерЃћрЃгрЃћрЃБрЃџрЃў рЃўрЃДрЃЮрЃА tab-рЃўрЃЌ 

function f1() {
    somecode(); // рЃерЃћрЃгрЃўрЃћ рЃћрЃарЃЌрЃў tab-рЃўрЃЌ. 
    if (somethingTrue()) {
        doStuff(); // рЃерЃћрЃгрЃўрЃћ рЃЎрЃўрЃЊрЃћрЃЋ рЃћрЃарЃЌрЃў tab-рЃўрЃЌ
    }
    someOtherCode();
}

рЃЎрЃЮрЃюрЃбрЃарЃЮрЃџрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃў

рЃњрЃљрЃЏрЃћрЃЮрЃарЃћрЃЉрЃљ - for loop

рЃЌрЃБ рЃњрЃЋрЃўрЃюрЃЊрЃљ рЃЎрЃљрЃарЃћрЃџрЃЏрЃљ рЃарЃљрЃЏрЃћ рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃљ (рЃљрЃю рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃћрЃЉрЃў) N-рЃ»рЃћрЃа рЃњрЃљрЃўрЃЏрЃћрЃЮрЃарЃЮрЃА, рЃљрЃЏ рЃарЃўрЃфрЃ«рЃЋрЃА рЃЋрЃгрЃћрЃарЃЌ рЃљрЃАрЃћрЃЌ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљрЃерЃў. рЃљрЃ«рЃџрЃљ рЃўрЃЦрЃюрЃћрЃЉрЃљ рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљ, рЃарЃЮрЃфрЃљ рЃњрЃћрЃбрЃДрЃЋрЃўрЃЌ рЃарЃЮрЃЏ рЃарЃљрЃдрЃљрЃф рЃюрЃљрЃгрЃўрЃџрЃА рЃљрЃа рЃЏрЃўрЃљрЃЦрЃфрЃўрЃЮрЃЌ рЃДрЃБрЃарЃљрЃЊрЃдрЃћрЃЉрЃљ рЃарЃЮрЃњрЃЮрЃа рЃЏрЃБрЃерЃљрЃЮрЃЉрЃА, рЃЌрЃБрЃЏрЃфрЃљ рЃљрЃарЃљ рЃЉрЃЮрЃџрЃЮ. рЃќрЃћрЃЊрЃЏрЃћрЃбрЃў рЃўрЃюрЃцрЃЮрЃарЃЏрЃљрЃфрЃўрЃўрЃА рЃњрЃљрЃцрЃўрЃџрЃбрЃЋрЃарЃљ рЃљрЃарЃўрЃА рЃБрЃюрЃљрЃарЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃћрЃ«рЃЏрЃљрЃарЃћрЃЉрЃљрЃЌ рЃњрЃљрЃљрЃЎрЃћрЃЌрЃЮрЃЌ рЃЏрЃћрЃбрЃў рЃърЃарЃЮрЃњрЃарЃћрЃАрЃў рЃЊрЃљ рЃЏрЃЮрЃњрЃЋрЃўрЃљрЃюрЃћрЃЉрЃўрЃЌ рЃЏрЃЮрЃБрЃЉрЃарЃБрЃюрЃЊрЃћрЃЌ рЃарЃЌрЃБрЃџ рЃюрЃљрЃгрЃўрЃџрЃћрЃЉрЃА. рЃљрЃАрЃћрЃЋрЃћ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃюрЃўрЃљ рЃўрЃА, рЃарЃЮрЃЏ рЃърЃарЃЮрЃњрЃарЃћрЃАрЃўрЃА рЃерЃћрЃЏрЃЊрЃћрЃњ рЃЏрЃћрЃбрЃў рЃерЃљрЃюрЃАрЃў рЃњрЃљрЃЦрЃЋрЃЌ рЃћрЃА рЃарЃЌрЃБрЃџрЃў рЃюрЃљрЃгрЃўрЃџрЃў рЃњрЃљрЃўрЃњрЃЮрЃЌ, рЃарЃљрЃЊрЃњрЃљрЃю рЃБрЃЎрЃЋрЃћ рЃЏрЃћрЃбрЃў рЃњрЃљрЃЏрЃЮрЃфрЃЊрЃўрЃџрЃћрЃЉрЃљ рЃњрЃљрЃЦрЃЋрЃЌ. рЃБрЃЉрЃарЃљрЃџрЃЮрЃЊ рЃЊрЃљрЃўрЃЏрЃљрЃ«рЃАрЃЮрЃЋрЃарЃћрЃЌ, рЃарЃЮрЃЏ рЃерЃБрЃљ рЃарЃўрЃфрЃ«рЃЋрЃў рЃБрЃюрЃЊрЃљ рЃерЃћрЃфрЃЋрЃљрЃџрЃЮрЃЌ. 

for (var i=0; i < 9; i++) {
    move(); // рЃњрЃљрЃЊрЃљрЃЋрЃљ 9-рЃ»рЃћрЃа
}

рЃњрЃљрЃЏрЃћрЃЮрЃарЃћрЃЉрЃўрЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљ рЃЉрЃћрЃЋрЃа рЃАрЃљрЃЦрЃЏрЃћрЃА рЃњрЃЋрЃўрЃЏрЃљрЃарЃбрЃўрЃЋрЃћрЃЉрЃА, рЃЏрЃљрЃњрЃарЃљрЃЏ рЃљрЃЏ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљрЃерЃўрЃф рЃЎрЃў рЃЕрЃЋрЃћрЃюрЃА рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљрЃА рЃљрЃарЃљрЃюрЃљрЃўрЃарЃў рЃўрЃюрЃбрЃћрЃџрЃћрЃЦрЃбрЃў рЃљрЃа рЃљрЃЦрЃЋрЃА. рЃерЃћрЃБрЃФрЃџрЃўрЃљ рЃерЃћрЃљрЃАрЃарЃБрЃџрЃЮрЃА рЃЏрЃерЃарЃљрЃџрЃљрЃЊ рЃЊрЃљрЃгрЃћрЃарЃўрЃџрЃў рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃћрЃЉрЃў рЃћрЃарЃЌрЃЏрЃљрЃюрЃћрЃЌрЃўрЃА рЃЏрЃўрЃДрЃЮрЃџрЃћрЃЉрЃўрЃЌ, рЃЏрЃљрЃњрЃарЃљрЃЏ рЃЏрЃљрЃА рЃљрЃа рЃљрЃЦрЃЋрЃА рЃњрЃљрЃЊрЃљрЃгрЃДрЃЋрЃћрЃбрЃўрЃџрЃћрЃЉрЃўрЃА рЃЏрЃўрЃдрЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃџрЃЮрЃЉрЃљ.

рЃърЃўрЃарЃЮрЃЉрЃўрЃА рЃерЃћрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃљ - if/else statement

if (somethingTrue()) {
    doStuff();
} else {
    doAnotherStuff();
}

while (somethingTrue()) {
    keepDoingStuff()
