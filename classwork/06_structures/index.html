рЃћрЃарЃЌ-рЃћрЃарЃЌрЃў, рЃарЃљрЃф рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃЊрЃљрЃЏрЃљрЃЉрЃюрЃћрЃЋрЃћрЃџрЃў рЃўрЃДрЃЮрЃА, рЃљрЃарЃўрЃА text рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃА рЃљрЃарЃњрЃБрЃЏрЃћрЃюрЃбрЃћрЃЉрЃў. рЃЊрЃљрЃцрЃўрЃЦрЃарЃЊрЃў, рЃарЃљрЃбрЃЮрЃЏ рЃЋрЃљрЃгрЃЋрЃЊрЃўрЃЌ рЃЏрЃљрЃА x-рЃА рЃЮрЃарЃ»рЃћрЃа?

рЃЏрЃўрЃюрЃўрЃерЃюрЃћрЃЉрЃљ: рЃАрЃфрЃљрЃЊрЃћ рЃљрЃАрЃћрЃЌрЃў рЃЎрЃЮрЃЊрЃў 

text("| " + x, x, 10);

рЃљрЃЏрЃЮрЃ«рЃАрЃюрЃўрЃА walkthrough

рЃЏрЃЮрЃЊрЃў рЃњрЃљрЃЊрЃљрЃЋрЃгрЃћрЃарЃЮрЃЌ рЃЌрЃљрЃЋрЃЊрЃљрЃърЃўрЃарЃЋрЃћрЃџрЃў рЃЎрЃЮрЃЊрЃў рЃўрЃАрЃћ, рЃарЃЮрЃЏ рЃарЃљрЃўрЃЏрЃћ рЃАрЃљрЃ«рЃўрЃА рЃњрЃљрЃЏрЃћрЃЮрЃарЃћрЃЉрЃљрЃЊрЃў рЃърЃљрЃбрЃћрЃарЃюрЃў рЃЏрЃўрЃЋрЃўрЃдрЃЮрЃЌ. 

var x = 0;
text(x, x, 10); 
x = x + 50; // рЃњрЃљрЃ«рЃЊрЃљ 50
text(x, x, 10);
x = x + 50; // рЃњрЃљрЃ«рЃЊрЃљ 100
text(x, x, 10);
x = x + 50; 
text(x, x, 10);
x = x + 50;
рЃАрЃ«рЃЋрЃљ рЃАрЃўрЃбрЃДрЃЋрЃћрЃЉрЃўрЃЌ рЃарЃЮрЃЏ рЃљрЃЋрЃдрЃгрЃћрЃарЃЮрЃЌ, x рЃЎрЃЮрЃЮрЃарЃЊрЃўрЃюрЃљрЃбрЃќрЃћ рЃЋрЃгрЃћрЃарЃЌ рЃбрЃћрЃЦрЃАрЃбрЃА, рЃЋрЃќрЃарЃЊрЃўрЃЌ 50-рЃўрЃЌ, рЃЊрЃљ рЃљрЃАрЃћ рЃЋрЃўрЃЏрЃћрЃЮрЃарЃћрЃЉрЃЌ рЃЏрЃљрЃюрЃљрЃЏ, рЃАрЃљрЃюрЃљрЃЏ x рЃљрЃарЃўрЃА 350-рЃќрЃћ рЃюрЃљрЃЎрЃџрЃћрЃЉрЃў.

рЃљрЃАрЃћ рЃЕрЃљрЃЏрЃЮрЃДрЃљрЃџрЃўрЃЉрЃћрЃЉрЃўрЃЌ рЃБрЃЎрЃЋрЃћ рЃфрЃўрЃЎрЃџрЃўрЃА рЃДрЃЋрЃћрЃџрЃљ рЃюрЃљрЃгрЃўрЃџрЃў рЃњрЃЋрЃљрЃЦрЃЋрЃА: 

var x = 0;
while (x < 400) {
    text(x, x, 10); 
    x = x + 50; 
}

рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃўрЃЌрЃў рЃърЃўрЃарЃЮрЃЉрЃљ

рЃ░рЃЮрЃарЃўрЃќрЃЮрЃюрЃбрЃљрЃџрЃБрЃарЃў/рЃЋрЃћрЃарЃбрЃўрЃЎрЃљрЃџрЃБрЃарЃўрЃА рЃЏрЃљрЃњрЃўрЃЋрЃарЃљрЃЊ, рЃЊрЃљрЃгрЃћрЃарЃћ рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃў рЃЊрЃўрЃљрЃњрЃЮрЃюрЃљрЃџрЃќрЃћ. 

var d = 0;
while (d < 400) {
    text("* " + d, d, d); 
    d = d + 50; 
}

рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃўрЃЌрЃў рЃърЃўрЃарЃЮрЃЉрЃљ (optional)

рЃЊрЃљрЃгрЃћрЃарЃћ рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃў рЃАрЃљрЃърЃўрЃарЃўрЃАрЃърЃўрЃарЃЮ рЃЊрЃўрЃљрЃњрЃЮрЃюрЃљрЃџрЃќрЃћ (рЃќрЃћрЃЊрЃљ рЃЏрЃљрЃарЃ»рЃЋрЃћрЃюрЃљ рЃЎрЃБрЃЌрЃ«рЃўрЃЊрЃљрЃю рЃЦрЃЋрЃћрЃЊрЃљ рЃЏрЃљрЃарЃфрЃ«рЃћрЃюрЃљрЃЏрЃЊрЃћ). 

var d = 0;
